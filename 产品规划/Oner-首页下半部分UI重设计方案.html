// ====== 1. StatusFilterBar.jsx ======
const statusFilters = [
  { value: '',      label: '全部',   emoji: '📋', color: 'all' },
  { value: 'note',    label: '备忘',   emoji: '📝', color: 'memo' },
  { value: 'todo',    label: '待办',   emoji: '⏰', color: 'todo' },
  { value: 'in_progress', label: '进行中', emoji: '🚀', color: 'progress' },
  { value: 'done',    label: '已完成', emoji: '✅', color: 'done' },
];

// 统计数量
const counts = {
  '': allNotes.length,
  note: allNotes.filter(n => n.status === 'note').length,
  todo: allNotes.filter(n => n.status === 'todo').length,
  in_progress: allNotes.filter(n => n.status === 'in_progress').length,
  done: allNotes.filter(n => n.status === 'done').length,
};

// 选中样式映射
const activeClassMap = {
  '': 'active-all', note: 'active-memo', todo: 'active-todo',
  in_progress: 'active-progress', done: 'active-done'
};

// 使用
<div className="status-filter-bar">
  {statusFilters.map(f => (
    <button
      key={f.value}
      onClick={() => handleStatusChange(f.value)}
      className={`status-pill ${activeStatus === f.value ? activeClassMap[f.value] : ''}`}
    >
      <span className="pill-icon">{f.emoji}</span>
      <span>{f.label}</span>
      <span className="pill-count">{counts[f.value]}</span>
    </button>
  ))}
</div>

// ====== 2. NoteCard 顶部色带增强 ======
// 在 NoteCard 的 className 中增加状态类：

};

<div className={`note-card ${statusTopClass[note.status] || ''}`}>
  // 内部结构保持不变，hover 时顶部色带自动显示
</div>
